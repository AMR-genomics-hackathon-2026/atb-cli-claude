# atb-cli Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone Go CLI that queries AllTheBacteria parquet files (~3.2M genomes) and downloads genome assemblies.

**Architecture:** Pure Go binary using `parquet-go` for parquet reading with struct-based column projection, `cobra` for CLI, `BurntSushi/toml` for config. Query engine filters `assembly.parquet` first, then joins secondary tables via `sample_accession` hash set. Downloads use parallel HTTP with resume support and disk space checks.

**Tech Stack:** Go 1.22+, parquet-go/parquet-go v0.29.0, spf13/cobra v1.10.2, BurntSushi/toml v1.6.0, schollz/progressbar/v3, olekukonez/tablewriter

```go
package config

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"github.com/BurntSushi/toml"
)

type Config struct {
	General  GeneralConfig  `toml:"general"`
	Fetch    FetchConfig    `toml:"fetch"`
	Download DownloadConfig `toml:"download"`
}

type GeneralConfig struct {
	DataDir       string `toml:"data_dir"`
	DefaultFormat string `toml:"default_format"`
}

type FetchConfig struct {
	BaseURL  string `toml:"base_url"`
	Parallel int    `toml:"parallel"`
}

type DownloadConfig struct {
	Parallel       int    `toml:"parallel"`
	OutputDir      string `toml:"output_dir"`
	CheckDiskSpace bool   `toml:"check_disk_space"`
	MinFreeSpaceGB int    `toml:"min_free_space_gb"`
}

func Default() *Config {
	home, _ := os.UserHomeDir()
	return &Config{
		General: GeneralConfig{
			DataDir:       filepath.Join(home, "atb", "metadata", "parquet"),
			DefaultFormat: "auto",
		},
		Fetch: FetchConfig{
			BaseURL:  "https://osf.io",
			Parallel: 4,
		},
		Download: DownloadConfig{
			Parallel:       4,
			OutputDir:      "./genomes",
			CheckDiskSpace: true,
			MinFreeSpaceGB: 5,
		},
	}
}

func DefaultPath() string {
	cfgDir, err := os.UserConfigDir()
	if err != nil {
		home, _ := os.UserHomeDir()
		cfgDir = filepath.Join(home, ".config")
	}
	return filepath.Join(cfgDir, "atb", "config.toml")
}

func Load(path string) (*Config, error) {
	cfg := Default()
	data, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return cfg, nil
		}
		return nil, fmt.Errorf("read config: %w", err)
	}
	if err := toml.Unmarshal(data, cfg); err != nil {
		return nil, fmt.Errorf("parse config %s: %w", path, err)
	}
	cfg.General.DataDir = ExpandPath(cfg.General.DataDir)
	return cfg, nil
}

func Save(cfg *Config, path string) error {
	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return fmt.Errorf("create config dir: %w", err)
	}
	f, err := os.Create(path)
	if err != nil {
		return fmt.Errorf("create config file: %w", err)
	}
	defer f.Close()
	enc := toml.NewEncoder(f)
	return enc.Encode(cfg)
}

func ExpandPath(path string) string {
	if strings.HasPrefix(path, "~/") {
		home, err := os.UserHomeDir()
		if err != nil {
			return path
		}
		return filepath.Join(home, path[2:])
	}
	return path
}
```

```go
package cli

import (
	"fmt"
	"os"
	"strings"

	"github.com/BurntSushi/toml"
	"github.com/AMR-genomics-hackathon-2026/atb-cli-claude/internal/config"
	"github.com/spf13/cobra"
)

func newConfigCmd() *cobra.Command {
	cmd := &cobra.Command{
		Use:   "config",
		Short: "Manage configuration",
	}
	cmd.AddCommand(newConfigInitCmd())
	cmd.AddCommand(newConfigShowCmd())
	cmd.AddCommand(newConfigSetCmd())
	cmd.AddCommand(newConfigGetCmd())
	return cmd
}

func newConfigInitCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "init",
		Short: "Create default config file",
		RunE: func(cmd *cobra.Command, args []string) error {
			path := resolveConfigPath()
			if _, err := os.Stat(path); err == nil {
				return fmt.Errorf("config already exists at %s (use --force to overwrite)", path)
			}
			cfg := config.Default()
			if err := config.Save(cfg, path); err != nil {
				return err
			}
			fmt.Fprintf(cmd.OutOrStdout(), "Config created at %s\n", path)
			return nil
		},
	}
}

func newConfigShowCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "show",
		Short: "Print current configuration",
		RunE: func(cmd *cobra.Command, args []string) error {
			cfg, err := loadConfig()
			if err != nil {
				return err
			}
			return toml.NewEncoder(cmd.OutOrStdout()).Encode(cfg)
		},
	}
}

func newConfigSetCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "set <key> <value>",
		Short: "Set a config value (e.g., atb config set general.data_dir /path)",
		Args:  cobra.ExactArgs(2),
		RunE: func(cmd *cobra.Command, args []string) error {
			path := resolveConfigPath()
			cfg, err := config.Load(path)
			if err != nil {
				return err
			}
			if err := setConfigValue(cfg, args[0], args[1]); err != nil {
				return err
			}
			if err := config.Save(cfg, path); err != nil {
				return err
			}
			fmt.Fprintf(cmd.OutOrStdout(), "Set %s = %s\n", args[0], args[1])
			return nil
		},
	}
}

func newConfigGetCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "get <key>",
		Short: "Get a config value",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			cfg, err := loadConfig()
			if err != nil {
				return err
			}
			val, err := getConfigValue(cfg, args[0])
			if err != nil {
				return err
			}
			fmt.Fprintln(cmd.OutOrStdout(), val)
			return nil
		},
	}
}

func resolveConfigPath() string {
	if cfgFile != "" {
		return cfgFile
	}
	return config.DefaultPath()
}

func loadConfig() (*config.Config, error) {
	return config.Load(resolveConfigPath())
}

func setConfigValue(cfg *config.Config, key, value string) error {
	switch strings.ToLower(key) {
	case "general.data_dir":
		cfg.General.DataDir = config.ExpandPath(value)
	case "general.default_format":
		cfg.General.DefaultFormat = value
	case "fetch.base_url":
		cfg.Fetch.BaseURL = value
	case "fetch.parallel":
		n, err := parseInt(value)
		if err != nil {
			return err
		}
		cfg.Fetch.Parallel = n
	case "download.parallel":
		n, err := parseInt(value)
		if err != nil {
			return err
		}
		cfg.Download.Parallel = n
	case "download.output_dir":
		cfg.Download.OutputDir = value
	case "download.min_free_space_gb":
		n, err := parseInt(value)
		if err != nil {
			return err
		}
		cfg.Download.MinFreeSpaceGB = n
	default:
		return fmt.Errorf("unknown config key: %s", key)
	}
	return nil
}

func getConfigValue(cfg *config.Config, key string) (string, error) {
	switch strings.ToLower(key) {
	case "general.data_dir":
		return cfg.General.DataDir, nil
	case "general.default_format":
		return cfg.General.DefaultFormat, nil
	case "fetch.base_url":
		return cfg.Fetch.BaseURL, nil
	case "fetch.parallel":
		return fmt.Sprintf("%d", cfg.Fetch.Parallel), nil
	case "download.parallel":
		return fmt.Sprintf("%d", cfg.Download.Parallel), nil
	case "download.output_dir":
		return cfg.Download.OutputDir, nil
	case "download.check_disk_space":
		return fmt.Sprintf("%t", cfg.Download.CheckDiskSpace), nil
	case "download.min_free_space_gb":
		return fmt.Sprintf("%d", cfg.Download.MinFreeSpaceGB), nil
	default:
		return "", fmt.Errorf("unknown config key: %s", key)
	}
}

func parseInt(s string) (int, error) {
	var n int
	_, err := fmt.Sscanf(s, "%d", &n)
	if err != nil {
		return 0, fmt.Errorf("invalid integer: %s", s)
	}
	return n, nil
}
```

```go
package parquet

import (
	"fmt"
	"io"
	"os"

	"github.com/parquet-go/parquet-go"
)

// ReadAll reads all rows from a parquet file into a slice of T.
// Only columns matching T's parquet tags are read (column projection).
func ReadAll[T any](path string) ([]T, error) {
	f, err := os.Open(path)
	if err != nil {
		return nil, fmt.Errorf("open %s: %w", path, err)
	}
	defer f.Close()

	stat, err := f.Stat()
	if err != nil {
		return nil, fmt.Errorf("stat %s: %w", path, err)
	}

	reader := parquet.NewGenericReader[T](f, parquet.ReadBufferSize(stat.Size()))
	defer reader.Close()

	rows := make([]T, reader.NumRows())
	n, err := reader.Read(rows)
	if err != nil && err != io.EOF {
		return nil, fmt.Errorf("read %s: %w", path, err)
	}
	return rows[:n], nil
}

// ReadFiltered reads rows from a parquet file, keeping only those where fn returns true.
func ReadFiltered[T any](path string, fn func(*T) bool) ([]T, error) {
	f, err := os.Open(path)
	if err != nil {
		return nil, fmt.Errorf("open %s: %w", path, err)
	}
	defer f.Close()

	stat, err := f.Stat()
	if err != nil {
		return nil, fmt.Errorf("stat %s: %w", path, err)
	}

	reader := parquet.NewGenericReader[T](f, parquet.ReadBufferSize(stat.Size()))
	defer reader.Close()

	const batchSize = 4096
	batch := make([]T, batchSize)
	var result []T

	for {
		n, err := reader.Read(batch)
		for i := 0; i < n; i++ {
			if fn(&batch[i]) {
				result = append(result, batch[i])
			}
		}
		if err != nil {
			if err == io.EOF {
				break
			}
			return nil, fmt.Errorf("read %s: %w", path, err)
		}
	}
	return result, nil
}
```

```go
package query

import (
	"fmt"
	"os"
	"strings"

	"github.com/BurntSushi/toml"
)

type FilterFile struct {
	Filter Filters      `toml:"filter"`
	Output OutputConfig `toml:"output"`
}

type Filters struct {
	Species            string  `toml:"species"`
	SpeciesLike        string  `toml:"species_like"`
	Genus              string  `toml:"genus"`
	Samples            []string `toml:"samples"`
	SampleFile         string  `toml:"sample_file"`
	HQOnly             bool    `toml:"hq_only"`
	MinCompleteness    float64 `toml:"min_completeness"`
	MaxContamination   float64 `toml:"max_contamination"`
	MinN50             int64   `toml:"min_n50"`
	Dataset            string  `toml:"dataset"`
	HasAssembly        bool    `toml:"has_assembly"`
	Country            string  `toml:"country"`
	Platform           string  `toml:"platform"`
	CollectionDateFrom string  `toml:"collection_date_from"`
	CollectionDateTo   string  `toml:"collection_date_to"`
}

type OutputConfig struct {
	Columns  []string `toml:"columns"`
	SortBy   string   `toml:"sort_by"`
	SortDesc bool     `toml:"sort_desc"`
	Limit    int      `toml:"limit"`
	Offset   int      `toml:"offset"`
	Format   string   `toml:"format"`
	Output   string   `toml:"output"`
}

func LoadFilterFile(path string) (*FilterFile, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read filter file: %w", err)
	}
	var ff FilterFile
	if err := toml.Unmarshal(data, &ff); err != nil {
		return nil, fmt.Errorf("parse filter file %s: %w", path, err)
	}
	return &ff, nil
}

func (f *Filters) NeedsCheckM2() bool {
	return f.MinCompleteness > 0 || f.MaxContamination > 0
}

func (f *Filters) NeedsAssemblyStats() bool {
	return f.MinN50 > 0
}

func (f *Filters) NeedsENA() bool {
	return f.Country != "" || f.Platform != "" || f.CollectionDateFrom != "" || f.CollectionDateTo != ""
}

func (f *Filters) NeedsSylph() bool {
	return false // reserved for future sylph-specific filters
}

func (f *Filters) MatchesSpecies(species string) bool {
	if f.Species == "" {
		return true
	}
	return strings.EqualFold(f.Species, species)
}

func (f *Filters) MatchesSpeciesLike(species string) bool {
	if f.SpeciesLike == "" {
		return true
	}
	pattern := strings.ToLower(f.SpeciesLike)
	value := strings.ToLower(species)

	if strings.HasPrefix(pattern, "%") && strings.HasSuffix(pattern, "%") {
		return strings.Contains(value, pattern[1:len(pattern)-1])
	}
	if strings.HasPrefix(pattern, "%") {
		return strings.HasSuffix(value, pattern[1:])
	}
	if strings.HasSuffix(pattern, "%") {
		return strings.HasPrefix(value, pattern[:len(pattern)-1])
	}
	return strings.EqualFold(f.SpeciesLike, species)
}

func (f *Filters) LoadSampleFile() error {
	if f.SampleFile == "" {
		return nil
	}
	data, err := os.ReadFile(f.SampleFile)
	if err != nil {
		return fmt.Errorf("read sample file: %w", err)
	}
	lines := strings.Split(strings.TrimSpace(string(data)), "\n")
	for _, line := range lines {
		line = strings.TrimSpace(line)
		if line != "" && !strings.HasPrefix(line, "#") {
			f.Samples = append(f.Samples, line)
		}
	}
	return nil
}

func (f *Filters) HasSampleFilter() bool {
	return len(f.Samples) > 0
}

func (f *Filters) SampleSet() map[string]struct{} {
	set := make(map[string]struct{}, len(f.Samples))
	for _, s := range f.Samples {
		set[s] = struct{}{}
	}
	return set
}
```

```go
// QueryPlan describes which tables to read and in what order.
type QueryPlan struct {
	Tables []string
}

// columnsRequiringCheckM2 are output columns that live in checkm2.parquet.
var columnsRequiringCheckM2 = map[string]bool{
	"Completeness_General":  true,
	"Contamination":         true,
	"Completeness_Specific": true,
	"Genome_Size":           true,
	"GC_Content":            true,

	"total_length": true,
	"number":       true,
	"mean_length":  true,
	"longest":      true,
	"shortest":     true,
	"N50":          true,
	"N90":          true,
}

// columnsRequiringENA are output columns from ena_20250506.parquet.

	"country":            true,
	"collection_date":    true,
	"instrument_platform": true,
	"instrument_model":   true,
	"read_count":         true,
	"base_count":         true,
	"library_strategy":   true,
	"study_accession":    true,
	"fastq_ftp":          true,
}

// columnsRequiringSylph are output columns from sylph.parquet.
var columnsRequiringSylph = map[string]bool{
	"Adjusted_ANI":        true,
	"Taxonomic_abundance": true,
	"Sequence_abundance":  true,
	"Median_cov":          true,
}

// Plan determines which tables are needed based on filters and requested columns.
func Plan(filters Filters, columns []string) QueryPlan {
	tables := []string{"assembly"}

	needCheckM2 := filters.NeedsCheckM2()
	needStats := filters.NeedsAssemblyStats()
	needENA := filters.NeedsENA()
	needSylph := filters.NeedsSylph()

	for _, col := range columns {
		if columnsRequiringCheckM2[col] {
			needCheckM2 = true
		}
		if columnsRequiringAssemblyStats[col] {
			needStats = true
		}
		if columnsRequiringENA[col] {
			needENA = true
		}
		if columnsRequiringSylph[col] {
			needSylph = true
		}
	}

	if needCheckM2 {
		tables = append(tables, "checkm2")
	}
	if needStats {
		tables = append(tables, "assembly_stats")
	}
	if needSylph {
		tables = append(tables, "sylph")
	}
	if needENA {
		tables = append(tables, "run", "ena_20250506")
	}

	return QueryPlan{Tables: tables}
}
```

```go
package query

import (
	"fmt"
	"path/filepath"
	"strings"

	pq "github.com/AMR-genomics-hackathon-2026/atb-cli-claude/internal/parquet"
)

// ResultRow is a map of column name to string value.
type ResultRow map[string]string

// Execute runs a query against the parquet files in dataDir.
func Execute(dataDir string, filters Filters, columns []string) ([]ResultRow, error) {
	if err := filters.LoadSampleFile(); err != nil {
		return nil, err
	}

	plan := Plan(filters, columns)

	// Step 1: Read and filter assembly.parquet
	assemblyPath := filepath.Join(dataDir, "assembly.parquet")
	sampleSet := filters.SampleSet()

	assemblies, err := pq.ReadFiltered[pq.AssemblyRow](assemblyPath, func(row *pq.AssemblyRow) bool {
		if filters.HQOnly && row.HQFilter != "PASS" {
			return false
		}
		if !filters.MatchesSpecies(row.SylphSpecies) {
			return false
		}
		if !filters.MatchesSpeciesLike(row.SylphSpecies) {
			return false
		}
		if filters.Genus != "" && !strings.EqualFold(pq.GenusFromSpecies(row.SylphSpecies), filters.Genus) {
			return false
		}
		if filters.Dataset != "" && row.Dataset != filters.Dataset {
			return false
		}
		if filters.HasAssembly && row.AsmFastaOnOSF != 1 {
			return false
		}
		if filters.HasSampleFilter() {
			if _, ok := sampleSet[row.SampleAccession]; !ok {
				return false
			}
		}
		return true
	})
	if err != nil {
		return nil, fmt.Errorf("read assembly: %w", err)
	}

	// Build lookup: sample_accession -> assembly data
	type sampleData struct {
		assembly pq.AssemblyRow
		checkm2  *pq.CheckM2Row
		stats    *pq.AssemblyStatsRow
		ena      *pq.ENARow
	}
	lookup := make(map[string]*sampleData, len(assemblies))
	for i := range assemblies {
		lookup[assemblies[i].SampleAccession] = &sampleData{assembly: assemblies[i]}
	}

	// Step 2: Join checkm2 if needed
	if containsTable(plan.Tables, "checkm2") {
		checkm2Path := filepath.Join(dataDir, "checkm2.parquet")
		rows, err := pq.ReadAll[pq.CheckM2Row](checkm2Path)
		if err != nil {
			return nil, fmt.Errorf("read checkm2: %w", err)
		}
		for i := range rows {
			if sd, ok := lookup[rows[i].SampleAccession]; ok {
				sd.checkm2 = &rows[i]
			}
		}
		// Filter by checkm2 criteria
		if filters.MinCompleteness > 0 || filters.MaxContamination > 0 {
			for key, sd := range lookup {
				if sd.checkm2 == nil {
					delete(lookup, key)
					continue
				}
				if filters.MinCompleteness > 0 && sd.checkm2.CompletenessGeneral < filters.MinCompleteness {
					delete(lookup, key)
					continue
				}
				if filters.MaxContamination > 0 && sd.checkm2.Contamination > filters.MaxContamination {
					delete(lookup, key)
				}
			}
		}
	}

	// Step 3: Join assembly_stats if needed
	if containsTable(plan.Tables, "assembly_stats") {
		statsPath := filepath.Join(dataDir, "assembly_stats.parquet")
		rows, err := pq.ReadAll[pq.AssemblyStatsRow](statsPath)
		if err != nil {
			return nil, fmt.Errorf("read assembly_stats: %w", err)
		}
		for i := range rows {
			if sd, ok := lookup[rows[i].SampleAccession]; ok {
				sd.stats = &rows[i]
			}
		}
		if filters.MinN50 > 0 {
			for key, sd := range lookup {
				if sd.stats == nil || sd.stats.N50 < filters.MinN50 {
					delete(lookup, key)
				}
			}
		}
	}

	// Step 4: Join ENA if needed
	if containsTable(plan.Tables, "ena_20250506") {
		// Build run_accession -> sample_accession mapping
		enaPath := filepath.Join(dataDir, "ena_20250506.parquet")
		enaRows, err := pq.ReadAll[pq.ENARow](enaPath)
		if err != nil {
			return nil, fmt.Errorf("read ena: %w", err)
		}
		for i := range enaRows {
			if sd, ok := lookup[enaRows[i].SampleAccession]; ok {
				sd.ena = &enaRows[i]
			}
		}
		// Filter by ENA criteria
		for key, sd := range lookup {
			if sd.ena == nil {
				if filters.NeedsENA() {
					delete(lookup, key)
				}
				continue
			}
			if filters.Country != "" && !strings.EqualFold(sd.ena.Country, filters.Country) {
				delete(lookup, key)
				continue
			}
			if filters.Platform != "" && !strings.EqualFold(sd.ena.InstrumentPlatform, filters.Platform) {
				delete(lookup, key)
				continue
			}
			if filters.CollectionDateFrom != "" && sd.ena.CollectionDate < filters.CollectionDateFrom {
				delete(lookup, key)
				continue
			}
			if filters.CollectionDateTo != "" && sd.ena.CollectionDate > filters.CollectionDateTo {
				delete(lookup, key)
			}
		}
	}

	// Step 5: Build result rows
	results := make([]ResultRow, 0, len(lookup))
	for _, sd := range lookup {
		row := ResultRow{
			"sample_accession": sd.assembly.SampleAccession,
			"run_accession":    sd.assembly.RunAccession,
			"sylph_species":    sd.assembly.SylphSpecies,
			"hq_filter":        sd.assembly.HQFilter,
			"dataset":          sd.assembly.Dataset,
			"aws_url":          sd.assembly.AWSUrl,
			"osf_tarball_url":  sd.assembly.OSFTarballURL,
		}
		if sd.checkm2 != nil {
			row["Completeness_General"] = fmt.Sprintf("%.1f", sd.checkm2.CompletenessGeneral)
			row["Contamination"] = fmt.Sprintf("%.1f", sd.checkm2.Contamination)
		}
		if sd.stats != nil {
			row["total_length"] = fmt.Sprintf("%d", sd.stats.TotalLength)
			row["N50"] = fmt.Sprintf("%d", sd.stats.N50)
			row["N90"] = fmt.Sprintf("%d", sd.stats.N90)
			row["number"] = fmt.Sprintf("%d", sd.stats.Number)
		}
		if sd.ena != nil {
			row["country"] = sd.ena.Country
			row["collection_date"] = sd.ena.CollectionDate
			row["instrument_platform"] = sd.ena.InstrumentPlatform
			row["instrument_model"] = sd.ena.InstrumentModel
		}
		results = append(results, row)
	}

	// Filter columns if specified
	if len(columns) > 0 {
		filtered := make([]ResultRow, len(results))
		for i, row := range results {
			newRow := make(ResultRow, len(columns))
			for _, col := range columns {
				if val, ok := row[col]; ok {
					newRow[col] = val
				}
			}
			filtered[i] = newRow
		}
		results = filtered
	}

	return results, nil
}

func containsTable(tables []string, name string) bool {
	for _, t := range tables {
		if t == name {
			return true
		}
	}
	return false
}
```

```go
	case "csv":
		return formatDelimited(w, rows, columns, ',')
	case "json":

	default:

func formatDelimited(w io.Writer, rows []query.ResultRow, columns []string, delim rune) error {
	writer := csv.NewWriter(w)
	writer.Comma = delim
	defer writer.Flush()

	if err := writer.Write(columns); err != nil {
		return err
	}
	for _, row := range rows {
		record := make([]string, len(columns))
		for i, col := range columns {
			record[i] = row[col]
		}
		if err := writer.Write(record); err != nil {
			return err
		}
	}
	return nil
}

func formatJSON(w io.Writer, rows []query.ResultRow, columns []string) error {
	output := make([]map[string]string, len(rows))
	for i, row := range rows {
		filtered := make(map[string]string, len(columns))
		for _, col := range columns {
			filtered[col] = row[col]
		}
		output[i] = filtered
	}
	enc := json.NewEncoder(w)
	enc.SetIndent("", "  ")
	return enc.Encode(output)
}

func formatTable(w io.Writer, rows []query.ResultRow, columns []string) error {
	table := tablewriter.NewWriter(w)
	headers := make([]string, len(columns))
	for i, col := range columns {
		headers[i] = strings.ToUpper(col)
	}
	table.SetHeader(headers)
	table.SetBorder(false)
	table.SetAutoWrapText(false)
	table.SetColumnSeparator(" ")

	for _, row := range rows {
		record := make([]string, len(columns))
		for i, col := range columns {
			record[i] = row[col]
		}
		table.Append(record)
	}

	fmt.Fprintf(w, "\n")
	table.Render()
	fmt.Fprintf(w, "\n%d results\n", len(rows))
	return nil
```

```go
package suggest

import (
	"sort"
	"strings"
)

func levenshtein(a, b string) int {
	la, lb := len(a), len(b)
	if la == 0 {
		return lb
	}
	if lb == 0 {
		return la
	}

	prev := make([]int, lb+1)
	curr := make([]int, lb+1)

	for j := 0; j <= lb; j++ {
		prev[j] = j
	}

	for i := 1; i <= la; i++ {
		curr[0] = i
		for j := 1; j <= lb; j++ {
			cost := 1
			if a[i-1] == b[j-1] {
				cost = 0
			}
			curr[j] = min(curr[j-1]+1, min(prev[j]+1, prev[j-1]+cost))
		}
		prev, curr = curr, prev
	}
	return prev[lb]
}

type scored struct {
	name  string
	score int
}

// Suggest returns up to n species names closest to the input.
func Suggest(input string, species []string, n int) []string {
	lower := strings.ToLower(input)
	candidates := make([]scored, 0, len(species))

	for _, s := range species {
		dist := levenshtein(lower, strings.ToLower(s))
		// Also check if input is a substring
		if strings.Contains(strings.ToLower(s), lower) {
			dist = dist / 2 // boost substring matches
		}
		candidates = append(candidates, scored{name: s, score: dist})
	}

	sort.Slice(candidates, func(i, j int) bool {
		return candidates[i].score < candidates[j].score
	})

	result := make([]string, 0, n)
	for i := 0; i < len(candidates) && i < n; i++ {
		result = append(result, candidates[i].name)
	}
	return result
}

func min(a, b int) int {
	if a < b {
		return a
	}
	return b
}
```

```go
package download

import (
	"fmt"
	"syscall"
)

// AvailableSpace returns available bytes on the filesystem containing path.
func AvailableSpace(path string) (uint64, error) {
	var stat syscall.Statfs_t
	if err := syscall.Statfs(path, &stat); err != nil {
		return 0, fmt.Errorf("statfs %s: %w", path, err)
	}
	return stat.Bavail * uint64(stat.Bsize), nil
}

// CheckDiskSpace verifies there's enough space for the download.
func CheckDiskSpace(dir string, neededBytes uint64, minFreeGB int) error {
	available, err := AvailableSpace(dir)
	if err != nil {
		return err
	}
	minFreeBytes := uint64(minFreeGB) * 1024 * 1024 * 1024
	if neededBytes+minFreeBytes > available {
		return fmt.Errorf(
			"insufficient disk space: need %s + %s reserve, only %s available",
			FormatBytes(neededBytes),
			FormatBytes(minFreeBytes),
			FormatBytes(available),
		)
	}
	return nil
}

// FormatBytes formats bytes as human-readable string.
func FormatBytes(bytes uint64) string {
	const (
		KB = 1024
		MB = KB * 1024
		GB = MB * 1024
		TB = GB * 1024
	)
	switch {
	case bytes >= TB:
		return fmt.Sprintf("%.1f TB", float64(bytes)/float64(TB))
	case bytes >= GB:
		return fmt.Sprintf("%.1f GB", float64(bytes)/float64(GB))
	case bytes >= MB:
		return fmt.Sprintf("%.1f MB", float64(bytes)/float64(MB))
	case bytes >= KB:
		return fmt.Sprintf("%.1f KB", float64(bytes)/float64(KB))
	default:
		return fmt.Sprintf("%d B", bytes)
	}
}
```

```go
type Config struct {
	OutputDir      string
	Parallel       int
	CheckDiskSpace bool
	MinFreeSpaceGB int
	MaxRetries     int
}

type Downloader struct {
	cfg    Config
	client *http.Client
}

type Result struct {
	Total     int
	Completed int
	Failed    int
	Bytes     int64
	Errors    []DownloadError
}

type DownloadError struct {
	URL   string
	Error string
}

type Manifest struct {
	Query     string    `json:"query"`
	Timestamp time.Time `json:"timestamp"`
	Total     int       `json:"total_files"`
	Completed int       `json:"completed"`
	Failed    int       `json:"failed"`
	TotalBytes int64    `json:"total_bytes"`
}

func New(cfg Config) *Downloader {
	if cfg.Parallel <= 0 {
		cfg.Parallel = 4
	}
	if cfg.MaxRetries <= 0 {
		cfg.MaxRetries = 3
	}
	return &Downloader{
		cfg: cfg,
		client: &http.Client{
			Timeout: 30 * time.Minute,
		},
	}
}

func (d *Downloader) DownloadFile(url, filename string) error {
	destPath := filepath.Join(d.cfg.OutputDir, filename)
	partPath := destPath + ".part"

	if err := os.MkdirAll(d.cfg.OutputDir, 0o755); err != nil {
		return fmt.Errorf("create output dir: %w", err)
	}

	// Check for existing complete file
	if _, err := os.Stat(destPath); err == nil {
		return nil // already downloaded
	}

	// Check for partial download (resume)
	var resumeFrom int64
	if info, err := os.Stat(partPath); err == nil {
		resumeFrom = info.Size()
	}

	req, err := http.NewRequest("GET", url, nil)
	if err != nil {
		return fmt.Errorf("create request: %w", err)
	}
	if resumeFrom > 0 {
		req.Header.Set("Range", fmt.Sprintf("bytes=%d-", resumeFrom))
	}

	var lastErr error
	for attempt := 0; attempt < d.cfg.MaxRetries; attempt++ {
		if attempt > 0 {
			time.Sleep(time.Duration(1<<uint(attempt)) * time.Second) // exponential backoff
		}

		resp, err := d.client.Do(req)
		if err != nil {
			lastErr = err
			continue
		}

		if resp.StatusCode == http.StatusRequestedRangeNotSatisfiable {
			resp.Body.Close()
			// File is complete, just rename
			resumeFrom = 0
			req.Header.Del("Range")
			continue
		}

		if resp.StatusCode != http.StatusOK && resp.StatusCode != http.StatusPartialContent {
			resp.Body.Close()
			lastErr = fmt.Errorf("HTTP %d for %s", resp.StatusCode, url)
			if resp.StatusCode == http.StatusTooManyRequests || resp.StatusCode >= 500 {
				continue // retry
			}
			return lastErr // non-retryable
		}

		flags := os.O_CREATE | os.O_WRONLY
		if resp.StatusCode == http.StatusPartialContent {
			flags |= os.O_APPEND
		} else {
			flags |= os.O_TRUNC
		}

		f, err := os.OpenFile(partPath, flags, 0o644)
		if err != nil {
			resp.Body.Close()
			return fmt.Errorf("open part file: %w", err)
		}

		_, err = io.Copy(f, resp.Body)
		f.Close()
		resp.Body.Close()

		if err != nil {
			lastErr = err
			continue
		}

		// Rename .part to final
		if err := os.Rename(partPath, destPath); err != nil {
			return fmt.Errorf("rename: %w", err)
		}
		return nil
	}
	return fmt.Errorf("download %s after %d retries: %w", url, d.cfg.MaxRetries, lastErr)
}

func (d *Downloader) DownloadAll(urls []string) Result {
	var result Result
	result.Total = len(urls)

	var mu sync.Mutex
	var completedCount atomic.Int32

	sem := make(chan struct{}, d.cfg.Parallel)
	var wg sync.WaitGroup

	for _, url := range urls {
		wg.Add(1)
		sem <- struct{}{}
		go func(u string) {
			defer wg.Done()
			defer func() { <-sem }()

			filename := filepath.Base(u)
			if err := d.DownloadFile(u, filename); err != nil {
				mu.Lock()
				result.Errors = append(result.Errors, DownloadError{URL: u, Error: err.Error()})
				mu.Unlock()
			} else {
				completedCount.Add(1)
			}
		}(url)
	}

	wg.Wait()
	result.Completed = int(completedCount.Load())
	result.Failed = result.Total - result.Completed
	return result
}

func (d *Downloader) WriteManifest(queryDesc string, result Result) error {
	manifest := Manifest{
		Query:      queryDesc,
		Timestamp:  time.Now().UTC(),
		Total:      result.Total,
		Completed:  result.Completed,
		Failed:     result.Failed,
		TotalBytes: result.Bytes,
	}
	path := filepath.Join(d.cfg.OutputDir, "manifest.json")
	f, err := os.Create(path)
	if err != nil {
		return err
	}
	defer f.Close()
	enc := json.NewEncoder(f)
	enc.SetIndent("", "  ")
	return enc.Encode(manifest)
}
```

```go
				urls = urls[:maxSamples]
			}

			if dryRun {
				fmt.Fprintf(cmd.OutOrStdout(), "Would download %d files to %s\n", len(urls), outputDir)
				for _, u := range urls {
					fmt.Fprintln(cmd.OutOrStdout(), u)
				}
				return nil
			}

			d := download.New(download.Config{
				OutputDir:      outputDir,
				Parallel:       parallel,
				CheckDiskSpace: cfg.Download.CheckDiskSpace && !force,
				MinFreeSpaceGB: cfg.Download.MinFreeSpaceGB,
			})

			fmt.Fprintf(cmd.ErrOrStderr(), "Downloading %d files to %s (parallel: %d)\n", len(urls), outputDir, parallel)
			result := d.DownloadAll(urls)

			fmt.Fprintf(cmd.ErrOrStderr(), "Completed: %d, Failed: %d\n", result.Completed, result.Failed)
			for _, e := range result.Errors {
				fmt.Fprintf(cmd.ErrOrStderr(), "  ERROR: %s: %s\n", filepath.Base(e.URL), e.Error)
			}

			return d.WriteManifest("download", result)
		},

func readURLFile(path string) ([]string, error) {
	f, err := os.Open(path)
	if err != nil {
		return nil, err
	}
	defer f.Close()

	var urls []string
	scanner := bufio.NewScanner(f)
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line != "" && !strings.HasPrefix(line, "#") {
			urls = append(urls, line)
		}
	}
	return urls, scanner.Err()
}

func extractURLsFromCSV(path, dataDir string) ([]string, error) {
	var r io.Reader
	if path == "-" {
		r = os.Stdin
	} else {
		f, err := os.Open(path)
		if err != nil {
			return nil, err
		}
		defer f.Close()
		r = f
	}

	reader := csv.NewReader(r)
	reader.LazyQuotes = true
	// Auto-detect delimiter
	reader.Comma = '\t'

	header, err := reader.Read()
	if err != nil {
		return nil, fmt.Errorf("read header: %w", err)
	}

	// If no tab-separated fields, try comma
	if len(header) == 1 && strings.Contains(header[0], ",") {
		// Re-parse as CSV - this is a simplified approach
		reader.Comma = ','
		parts := strings.Split(header[0], ",")
		header = parts
	}

	urlCol := -1
	sampleCol := -1
	for i, h := range header {
		h = strings.TrimSpace(h)
		if h == "aws_url" {
			urlCol = i
		}
		if h == "sample_accession" {
			sampleCol = i
		}
	}

	if urlCol == -1 && sampleCol == -1 {
		return nil, fmt.Errorf("CSV must contain 'aws_url' or 'sample_accession' column")
	}

	var urls []string
	for {
		record, err := reader.Read()
		if err == io.EOF {
			break
		}
		if err != nil {
			continue // skip malformed rows
		}

		if urlCol >= 0 && urlCol < len(record) && record[urlCol] != "" {
			urls = append(urls, strings.TrimSpace(record[urlCol]))
		}
	}

	return urls, nil
}
```

```go
func CoreTables() []string {
	return []string{
		"assembly.parquet",
		"assembly_stats.parquet",
		"checkm2.parquet",
		"sylph.parquet",
		"run.parquet",
	}

func AllTables() []string {
	tables := CoreTables()
	tables = append(tables,
		"ena_20250506.parquet",
		"ena_20240801.parquet",
		"ena_20240625.parquet",
		"ena_202505_used.parquet",
		"ena_661k.parquet",
	)
	return tables
}

func New(cfg Config) *Fetcher {
	return &Fetcher{
		cfg: cfg,
		client: &http.Client{
			Timeout: 60 * time.Minute,
		},
	}
}

func (f *Fetcher) FetchTable(name, url string) error {
	destPath := filepath.Join(f.cfg.DataDir, name)

	if err := os.MkdirAll(f.cfg.DataDir, 0o755); err != nil {
		return fmt.Errorf("create data dir: %w", err)
	}

	resp, err := f.client.Get(url)
	if err != nil {
		return fmt.Errorf("download %s: %w", name, err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return fmt.Errorf("download %s: HTTP %d", name, resp.StatusCode)
	}

	tmpPath := destPath + ".tmp"
	out, err := os.Create(tmpPath)
	if err != nil {
		return fmt.Errorf("create %s: %w", tmpPath, err)
	}

	_, err = io.Copy(out, resp.Body)
	out.Close()
	if err != nil {
		os.Remove(tmpPath)
		return fmt.Errorf("write %s: %w", name, err)
	}

	return os.Rename(tmpPath, destPath)
}

func (f *Fetcher) GetTableURL(name string) string {
	if url, ok := tableURLs[name]; ok {
		return url
	}
	return ""
}
```

```go
package summarise

import (
	"fmt"
	"io"
	"sort"

	"github.com/AMR-genomics-hackathon-2026/atb-cli-claude/internal/query"
)

type Summary struct {
	TotalGenomes int
	HQGenomes    int
	TopSpecies   []SpeciesCount
	Datasets     []DatasetCount
}

type SpeciesCount struct {
	Name  string
	Count int
}

type DatasetCount struct {
	Name  string
	Count int
}

type GroupCount struct {
	Key   string
	Count int
}

func DefaultSummary(rows []query.ResultRow) Summary {
	var s Summary
	s.TotalGenomes = len(rows)

	speciesMap := make(map[string]int)
	datasetMap := make(map[string]int)

	for _, row := range rows {
		if row["hq_filter"] == "PASS" {
			s.HQGenomes++
		}
		speciesMap[row["sylph_species"]]++
		datasetMap[row["dataset"]]++
	}

	for name, count := range speciesMap {
		s.TopSpecies = append(s.TopSpecies, SpeciesCount{Name: name, Count: count})
	}
	sort.Slice(s.TopSpecies, func(i, j int) bool {
		return s.TopSpecies[i].Count > s.TopSpecies[j].Count
	})

	for name, count := range datasetMap {
		s.Datasets = append(s.Datasets, DatasetCount{Name: name, Count: count})
	}
	sort.Slice(s.Datasets, func(i, j int) bool {
		return s.Datasets[i].Count > s.Datasets[j].Count
	})

	return s
}

func GroupBy(rows []query.ResultRow, dimension string) []GroupCount {
	counts := make(map[string]int)
	for _, row := range rows {
		key := row[dimension]
		if key == "" {
			key = "(unknown)"
		}
		counts[key]++
	}

	result := make([]GroupCount, 0, len(counts))
	for k, v := range counts {
		result = append(result, GroupCount{Key: k, Count: v})
	}
	sort.Slice(result, func(i, j int) bool {
		return result[i].Count > result[j].Count
	})
	return result
}

func PrintSummary(w io.Writer, s Summary, topN int) {
	fmt.Fprintf(w, "AllTheBacteria Database Summary\n")
	fmt.Fprintf(w, "==========================================\n\n")
	fmt.Fprintf(w, "Total genomes:      %d\n", s.TotalGenomes)
	if s.TotalGenomes > 0 {
		pct := float64(s.HQGenomes) / float64(s.TotalGenomes) * 100
		fmt.Fprintf(w, "  HQ genomes:       %d (%.1f%%)\n", s.HQGenomes, pct)
	}

	fmt.Fprintf(w, "\nTop %d species:\n", topN)
	limit := topN
	if limit > len(s.TopSpecies) {
		limit = len(s.TopSpecies)
	}
	for i := 0; i < limit; i++ {
		fmt.Fprintf(w, "  %-40s %d\n", s.TopSpecies[i].Name, s.TopSpecies[i].Count)
	}

	fmt.Fprintf(w, "\nDatasets:\n")
	for _, d := range s.Datasets {
		fmt.Fprintf(w, "  %-40s %d\n", d.Name, d.Count)
	}
}

func PrintGroupBy(w io.Writer, groups []GroupCount, dimension string, topN int) {
	fmt.Fprintf(w, "Group by: %s\n", dimension)
	fmt.Fprintf(w, "==========================================\n")
	limit := topN
	if limit <= 0 || limit > len(groups) {
		limit = len(groups)
	}
	for i := 0; i < limit; i++ {
		fmt.Fprintf(w, "  %-40s %d\n", groups[i].Key, groups[i].Count)
	}
	fmt.Fprintf(w, "\nTotal groups: %d\n", len(groups))
}
```
